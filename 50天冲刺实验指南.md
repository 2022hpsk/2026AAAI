### 3.3 任务 C：实现 speaker-aware 数据预处理

```python
class SpeakerAwareConverter:
    """
    把 EverMemBench / GroupMemBench / SocialMemBench 格式转换为
    SpeakerMem-R1 训练格式
    """
    
    def convert_conversation(self, raw_conv):
        """
        输入：GroupChatFormat 原始对话
        输出：SpeakerMem 训练样本
        """
        return {
            'conversation_id': raw_conv['group_id'],
            'speakers': [u['user_id'] for u in raw_conv['users']],
            'speaker_profiles': {
                u['user_id']: {
                    'name': u['name'],
                    'role': u.get('role', 'member')
                }
                for u in raw_conv['users']
            },
            'turns': [
                {
                    'turn_id': msg['message_id'],
                    'speaker_id': msg['sender'],
                    'content': msg['content'],
                    'timestamp': msg.get('timestamp', '')
                }
                for msg in raw_conv['conversation']
            ],
            'qa_pairs': raw_conv.get('qa_pairs', [])
        }
    
    def create_rl_training_sample(self, conversation, qa):
        """
        为 RL 训练构造 (system_prompt, user_prompt, gold_memory_state, gold_answer) 四元组
        """
        system_prompt = self.build_memory_agent_system_prompt(
            conversation['speakers'],
            conversation['speaker_profiles']
        )
        
        # 构造 gold memory state（ground truth memory 应该是什么）
        gold_memory = self.extract_gold_memory(conversation, qa)
        
        return {
            'system': system_prompt,
            'conversation': conversation['turns'],
            'question': qa['question'],
            'gold_memory': gold_memory,
            'gold_answer': qa['answer'],
            'asker_id': qa.get('asker_id', conversation['speakers'][0])
        }
    
    def build_memory_agent_system_prompt(self, speakers, profiles):
        speaker_descs = "\n".join([
            f"- {profiles[s]['name']} ({s}): {profiles[s].get('role', 'member')}"
            for s in speakers
        ])
        return f"""你是一个多方对话记忆助手。
当前群聊成员：
{speaker_descs}

你的任务是在处理群聊对话的同时，管理以下层级的记忆：
1. per_speaker_core[speaker_id]：每个说话人的核心事实
2. per_speaker_episodic[speaker_id]：每个说话人的情节记忆
3. per_speaker_profile[speaker_id]：每个说话人的个人画像
4. group_interaction：群组互动记录
5. group_insight：群组洞察

对每条新消息，你可以执行以下操作（JSON格式）：
{{"action": "WRITE", "content": "...", "speaker_id": "...", "audience": [...], "layer": "..."}}
{{"action": "UPDATE", "fragment_id": "...", "new_content": "..."}}
{{"action": "DELETE", "fragment_id": "..."}}
{{"action": "NOOP"}}

请始终保持记忆的准确性和说话人归属的正确性。"""
```

```python
from transformers import Trainer, TrainingArguments

def compute_role_masked_loss(model, inputs):
    """
    只对 memory agent 角色的 token 计算 loss，
    忽略用户/说话人输入的 token
    """
    labels = inputs['input_ids'].clone()
    
    # 找到 memory agent 开始生成的位置
    # （system prompt 之后，conversation 结束的位置）
    agent_start = inputs['agent_token_start']
    labels[:, :agent_start] = -100  # mask 掉非 agent 部分
    
    return model(**inputs, labels=labels).loss
```

### 4.2 Speaker-Aware Levenshtein Reward 实现

```python
import numpy as np
from scipy.optimize import linear_sum_assignment
from sentence_transformers import SentenceTransformer

class SpeakerLevenshteinReward:
    def __init__(self, tau=0.6, embedding_model='all-MiniLM-L6-v2'):
        self.tau = tau
        self.encoder = SentenceTransformer(embedding_model)
    
    def compute(self, pred_memory, gt_memory, speakers):
        """
        Speaker-aware Levenshtein F1 reward
        
        pred_memory: Dict[speaker_id, List[str]]  # 预测的每个 speaker 的 memory
        gt_memory: Dict[speaker_id, List[str]]    # ground truth
        speakers: List[str]
        """
        per_speaker_f1 = []
        
        for speaker in speakers:
            pred_frags = pred_memory.get(speaker, [])
            gt_frags = gt_memory.get(speaker, [])
            
            if not gt_frags:
                continue
            
            # 计算 embedding
            pred_embs = self.encoder.encode(pred_frags) if pred_frags else np.zeros((0, 384))
            gt_embs = self.encoder.encode(gt_frags)
            
            if len(pred_frags) == 0:
                per_speaker_f1.append(0.0)
                continue
            
            # 计算 similarity matrix
            sim_matrix = np.dot(pred_embs, gt_embs.T) / (
                np.linalg.norm(pred_embs, axis=1, keepdims=True) * 
                np.linalg.norm(gt_embs, axis=1)
            )
            
            # 最优匹配（Hungarian algorithm）
            row_ind, col_ind = linear_sum_assignment(-sim_matrix)  # 最大化
            
            # 过滤低相似度匹配（tau 阈值）
            matched_pairs = [
                (r, c) for r, c in zip(row_ind, col_ind)
                if sim_matrix[r, c] >= self.tau
            ]
            
            # 计算 lexical fidelity（可选）
            def lexical_check(pred_text, gt_text):
                gt_words = set(gt_text.lower().split())
                pred_words = set(pred_text.lower().split())
                return len(gt_words & pred_words) / len(gt_words) if gt_words else 0
            
            # soft precision / recall
            n_matched = len(matched_pairs)
            soft_precision = n_matched / len(pred_frags) if pred_frags else 0
            soft_recall = n_matched / len(gt_frags) if gt_frags else 0
            
            if soft_precision + soft_recall > 0:
                f1 = 2 * soft_precision * soft_recall / (soft_precision + soft_recall)
            else:
                f1 = 0.0
            
            per_speaker_f1.append(f1)
        
        return np.mean(per_speaker_f1) if per_speaker_f1 else 0.0
    
    def compute_cross_speaker_leakage(self, pred_memory, gt_memory, speakers):
        """
        计算 cross-speaker contamination（张冠李戴惩罚）
        """
        leakage = 0.0
        total_pairs = 0
        
        for s1 in speakers:
            for s2 in speakers:
                if s1 == s2:
                    continue
                
                # pred_memory[s1] 中是否混入了 gt_memory[s2] 的信息？
                pred_s1 = pred_memory.get(s1, [])
                gt_s2 = gt_memory.get(s2, [])
                
                if not pred_s1 or not gt_s2:
                    continue
                
                pred_embs = self.encoder.encode(pred_s1)
                gt_embs = self.encoder.encode(gt_s2)
                
                sim = np.max(np.dot(pred_embs, gt_embs.T) / (
                    np.linalg.norm(pred_embs, axis=1, keepdims=True) * 
                    np.linalg.norm(gt_embs, axis=1)
                ))
                
                if sim > 0.8:  # 高度相似 = 可能混入了其他人的信息
                    leakage += sim - 0.8  # 超出 0.8 的部分算 penalty
                
                total_pairs += 1
        
        return leakage / total_pairs if total_pairs > 0 else 0.0
```

### 4.3 完整 Reward Function 实现

```python
class SpeakerMemRewardV3:
    def __init__(self, llm_judge_model="gpt-4o-mini"):
        self.slr = SpeakerLevenshteinReward(tau=0.6)
        self.llm_judge = LLMJudge(llm_judge_model)
    
    def compute(self, trajectory, ground_truth, speakers):
        # 1. Task reward（token-level F1）
        R_task = token_f1(
            trajectory.final_answer, 
            ground_truth.gold_answer
        )
        
        # 2. Speaker attribution reward
        R_attr = self.slr.compute(
            pred_memory=trajectory.final_memory.per_speaker,
            gt_memory=ground_truth.memory.per_speaker,
            speakers=speakers
        )
        
        # 3. Speaker-aware Levenshtein state diff
        R_state = R_attr - 0.3 * self.slr.compute_cross_speaker_leakage(
            trajectory.final_memory.per_speaker,
            ground_truth.memory.per_speaker,
            speakers
        )
        
        # 4. Audience adaptation reward（LLM judge）
        R_aud = self.llm_judge.score_audience_adaptation(
            trajectory.outputs,
            ground_truth.audience_profiles,
            speakers
        )
        
        # 5. Compression reward
        R_compr = max(0, 1.0 - len(trajectory.final_memory) / ground_truth.max_memory_size)
        
        # 组合（权重参考 DeltaMem：state reward 权重大）
        total = (
            1.0 * R_task +
            0.5 * R_attr +
            0.8 * R_state +    # 主要 dense signal
            0.3 * R_aud +
            0.1 * R_compr
        )
        
        return total, {
            'R_task': R_task, 'R_attr': R_attr, 'R_state': R_state,
            'R_aud': R_aud, 'R_compr': R_compr
        }
```

### 5.2 Stage 3：Full LoGo-GRPO

```python
class SpeakerConditionedLoGoGRPO:
    """
    Speaker-conditioned Local + Global GRPO
    Memory-R2 的 LoGo-GRPO 扩展版本
    """
    def __init__(self, G_global=8, G_local=4, lambda_=0.5):
        self.G_global = G_global
        self.G_local = G_local
        self.lambda_ = lambda_
    
    def compute_loss(self, policy, memory_state, conversation, speakers, reward_fn):
        # === Global objective ===
        # 生成 G_global 条完整轨迹
        global_rollouts = [
            policy.rollout(memory_state, conversation, speakers)
            for _ in range(self.G_global)
        ]
        global_rewards = [reward_fn(r) for r in global_rollouts]
        global_advantage = self._group_relative_advantage(global_rewards)
        global_loss = self._policy_loss(policy, global_rollouts, global_advantage)
        
        # === Local objective（speaker-conditioned）===
        # 找到对话的中间点
        t_split = len(conversation) // 2
        shared_intermediate_state = simulate_to(
            policy, memory_state, conversation[:t_split], speakers
        )
        
        # 按 speaker subset 分组（每次固定一组 speaker 子集）
        local_loss = 0.0
        speaker_subsets = self._stratify_speaker_subsets(speakers)
        
        for subset in speaker_subsets:
            # 从 shared intermediate state 出发，只处理 subset 的 speaker turns
            local_rollouts = [
                policy.rollout(
                    shared_intermediate_state,
                    [t for t in conversation[t_split:] if t['speaker_id'] in subset],
                    subset
                )
                for _ in range(self.G_local)
            ]
            local_rewards = [reward_fn(r) for r in local_rollouts]
            local_advantage = self._group_relative_advantage(local_rewards)
            local_loss += self._policy_loss(policy, local_rollouts, local_advantage)
        
        local_loss /= len(speaker_subsets)
        
        # 组合
        total_loss = global_loss + self.lambda_ * local_loss
        return total_loss
    
    def _group_relative_advantage(self, rewards):
        rewards = np.array(rewards)
        return (rewards - rewards.mean()) / (rewards.std() + 1e-8)
    
    def _stratify_speaker_subsets(self, speakers):
        """
        对于大群聊（>6人），不穷举所有子集，
        而是按角色/频次采样 4-6 个代表性子集
        """
        if len(speakers) <= 4:
            return [speakers]  # 小群聊直接用全集
        else:
            # 随机采样 3 个子集（每个包含 3-5 个 speaker）
            subsets = []
            for _ in range(3):
                size = np.random.randint(3, min(6, len(speakers)))
                subset = np.random.choice(speakers, size=size, replace=False).tolist()
                subsets.append(subset)
            return subsets
```

### 7.3 风险 3：没有明显数字提升（中风险）

**问题**：相比 adapted baselines 提升不显著  
**应对**：
- 分析 error case：哪些类型的问题我们能明显改善？（speaker attribution 类应该强）
